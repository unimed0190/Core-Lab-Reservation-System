/*
 * style.css - 網站的通用樣式表
 * * 注意: 圖片路徑是相對於這個 CSS 檔案的位置。
 * 從 static/css/ 向上回到 static/ (使用 ../)
 * 然後進入 images/ 資料夾。
 */

/* ================================== */
/* 網站通用 Body 樣式 */
/* ================================== */

body {
    /* 預設字體和基本設置 */
    font-family: Arial, sans-serif;
    padding-top: 56px; /* 假設導航列的高度，避免內容被遮擋 */
    color: #333;
    min-height: 100vh; /* 確保 body 至少佔滿整個視窗高度 */

    /* 🌟 背景圖片設定 (請將 home_background.jpg 放在 static/images/ 中) 🌟 */
    background-image: url('../images/home_background.jpg'); 
    background-size: cover;       /* 使圖片覆蓋整個背景 */
    background-repeat: no-repeat; /* 不重複平鋪 */
    background-attachment: fixed; /* 讓背景圖片固定不動 */
    background-position: center center; /* 圖片置中 */
    
    /* 以防圖片無法載入時顯示的顏色 */
    background-color: #f8f9fa; 
}

/* ================================== */
/* 容器樣式 (讓內容在背景上可讀) */
/* ================================== */

.container {
    /* 內容容器使用微透明的白色，以確保文字在圖片背景上清晰可讀 */
    background-color: rgba(255, 255, 255, 0.9); 
    padding: 30px;
    border-radius: 10px;
    box-shadow: 0 4px 8px rgba(0, 0, 0, 0.1); /* 增加輕微陰影 */
    margin-top: 30px;
    margin-bottom: 30px;
}

/* 如果您不希望導航列有透明度，這裡可以調整 */
.navbar {
    opacity: 0.95; 
}
